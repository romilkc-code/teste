<script>

    document.getElementById('csvFile').addEventListener('change', function(e) {

        const file = e.target.files[0];

        if (!file) return;


        Papa.parse(file, {

            header: true,

            skipEmptyLines: true,

            complete: function(results) {

                processData(results.data);

            }

        });

    });


    function processData(data) {

        const monthlyData = {};

        let totalGlobal = 0;


        data.forEach(row => {

            // A DEGIRO identifica dividendos no campo 'Produto' ou se houver retenção na fonte.

            // Nota: Dependendo da versão do CSV, o filtro pode ser por "Dividendo" ou "Dividendos"

            // Aqui vamos filtrar se o valor total for positivo e não for uma venda (simplificação)

            // Ou melhor: procurar por termos específicos na descrição se disponível.

            

            const valorTotal = parseFloat(row['Total EUR']?.replace(',', '.') || 0);

            const dataBruta = row['Data']; // Formato DD-MM-YYYY

            

            // Lógica: Dividendos na DEGIRO costumam ser entradas de dinheiro (positivas) 

            // que não são depósitos. Vamos filtrar por valores positivos no "Total EUR".

            if (valorTotal > 0 && dataBruta) {

                const partes = dataBruta.split('-');

                const mesAno = partes[1] + '/' + partes[2]; // Formato MM/YYYY


                monthlyData[mesAno] = (monthlyData[mesAno] || 0) + valorTotal;

                totalGlobal += valorTotal;

            }

        });


        updateDashboard(totalGlobal, monthlyData);

    }


    let myChart;

    function updateDashboard(total, monthlyData) {

        document.getElementById('totalValue').innerText = total.toFixed(2) + '€';


        const ctx = document.getElementById('dividendChart').getContext('2d');

        const labels = Object.keys(monthlyData).reverse();

        const values = Object.values(monthlyData).reverse();


        if (myChart) myChart.destroy();


        myChart = new Chart(ctx, {

            type: 'bar',

            data: {

                labels: labels,

                datasets: [{

                    label: 'Dividendos Mensais (€)',

                    data: values,

                    backgroundColor: '#2ecc71',

                    borderRadius: 5

                }]

            },

            options: {

                scales: { y: { beginAtZero: true } }

            }

        });

    }

</script>
